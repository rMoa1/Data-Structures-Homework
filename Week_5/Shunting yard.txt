	1)How to postfix infix expressions with the Shunting Yard Algorithm? Example: a*b+c and (a+b)*c
#include <stdio.h>
#include <stdlib.h>
#include <ctype.h>

#define STACK_SIZE 20

typedef struct {
    int data[STACK_SIZE];
    int top;
} stack;

void initialize(stack * stk) {
    stk->top=-1;
}

void push (stack *stk, int c) {
    if (stk->top >= STACK_SIZE - 1) {
        printf("\nStack full\n\n");
        return;
    } else {
        stk->data[++stk->top] = c;
    }
}

int pop(stack *stk) {
    if (stk->top == -1) {
        printf("\nStack empty (underflow)\n\n");
        return 0; /* or handle error */
    }
    return stk->data[stk->top--];
}

int main () {
    char exp[] = "231*+9";

    char *e;
    stack s;
    initialize(&s);
    int n1, n2, n3, num;

    e=exp;

    while (*e != '\0') {
        if (isdigit((unsigned char)*e)) {
            num = *e - 48; 
            push(&s, num);
        } else {
            n1 = pop(&s);
            n2 = pop(&s);
            switch (*e) {
                case '+':
                    n3 = n2 + n1;
                    break;
                case '-':
                    n3 = n2 - n1;
                    break;
                case '*':
                    n3 = n2 * n1;
                    break;
                case '/':
                    if (n1 == 0) {
                        printf("Division by zero\n");
                        n3 = 0;
                    } else {
                        n3 = n2 / n1;
                    }
                    break;
                default:
                    printf("Unknown operator: %c\n", *e);
                    n3 = 0;
            }
            push(&s, n3);
            printf("operation: %d %c %d = %d\n", n2, *e, n1, n3);
        }
        e++; /* advance to next character */
    }

    /* final result (if any) */
    printf("Result = %d\n", pop(&s));
    

    return 0;
}
