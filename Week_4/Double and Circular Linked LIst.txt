	3)How you can add an element between elements at double linked list?
void insertBetween(struct Node *head, int key, int data) {
    struct Node *A = head;   
    if (A == NULL) {
        return;
    }    
    while (A != NULL && A->data != key) {
        A = A->next;
    }
    struct Node *N = createNode(data);
    struct Node *B = A->next;
    N->next = B;
    N->prev = A;
    A->next = N;
    }
}
