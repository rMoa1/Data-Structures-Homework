#include <stdio.h>
#include <ctype.h>

char stack[100];
int top = -1;

void push(char x) { stack[++top] = x; }
char pop() { return stack[top--]; }

int pr(char x) {
    if (x == '+' || x == '-') return 1;
    if (x == '*' || x == '/') return 2;
    if (x == '^') return 3;
    return 0;
}

int main() {
    char exp[100], *e, x;
    printf("Infix: ");
    scanf("%s", exp);
    e = exp;
    
    while (*e != '\0') {
        if (isalnum(*e)) printf("%c", *e);
        else if (*e == '(') push(*e);
        else if (*e == ')') {
            while ((x = pop()) != '(') printf("%c", x);
        } else {
            while (top != -1 && pr(stack[top]) >= pr(*e)) printf("%c", pop());
            push(*e);
        }
        e++;
    }
    while (top != -1) printf("%c", pop());
    printf("\n");
    return 0;
}
